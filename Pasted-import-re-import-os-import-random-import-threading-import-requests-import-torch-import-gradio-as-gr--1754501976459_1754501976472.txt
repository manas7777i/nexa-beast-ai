# Enhanced logging with rotation and file management
class SizeRotatingHandler(logging.FileHandler):
    def __init__(self, filename, max_bytes=10*1024*1024, backup_count=5, encoding=None, delay=False):
        self.max_bytes = max_bytes
        self.backup_count = backup_count
        self.current_size = 0
        super().__init__(filename, 'a', encoding, delay)
        self.check_size()

    def check_size(self):
        if self.stream:
            self.current_size = self.stream.tell()
            if self.current_size >= self.max_bytes:
                self.rotate()

    def rotate(self):
        self.close()
        for i in range(self.backup_count - 1, 0, -1):
            sfn = f"{self.baseFilename}.{i}"
            dfn = f"{self.baseFilename}.{i + 1}"
            if os.path.exists(sfn):
                if os.path.exists(dfn):
                    os.remove(dfn)
                os.rename(sfn, dfn)
        dfn = f"{self.baseFilename}.1"
        if os.path.exists(dfn):
            os.remove(dfn)
        os.rename(self.baseFilename, dfn)
        self.stream = self._open()

    def emit(self, record):
        super().emit(record)
        self.check_size()

class EnhancedRateLimiter:
    def __init__(self, max_calls: int, period: float):
        self.max_calls = max_calls
        self.period = period
        self.calls = []
        self.lock = Lock()

    def __call__(self, func):
        def wrapped(*args, **kwargs):
            with self.lock:
                now = time.time()
                self.calls = [call for call in self.calls if call > now - self.period]
                if len(self.calls) >= self.max_calls:
                    sleep_time = self.period - (now - self.calls[0])
                    time.sleep(sleep_time)
                self.calls.append(time.time())
            return func(*args, **kwargs)
        return wrapped

def rate_limited(max_concurrent=10, per_second=5):
    """Enhanced rate limiting decorator with both concurrent and per-second limits"""
    semaphore = Semaphore(max_concurrent)
    limiter = EnhancedRateLimiter(per_second, 1.0)
    
    def decorator(f):
        def wrapped(*args, **kwargs):
            with semaphore:
                return limiter(f)(*args, **kwargs)
        return wrapped
    return decorator

def clean_text(text: str, aggressive=False) -> str:
    """Enhanced text cleaning and normalization with multiple levels"""
    if not text:
        return ""
        
    # Basic cleaning
    text = html.unescape(text)
    text = re.sub(r'<[^>]+>', '', text)  # Remove HTML tags
    text = re.sub(r'http[s]?://(?:[a-zA-Z]|[0-9]|[$-_@.&+]|[!*\\(\\),]|(?:%[0-9a-fA-F][0-9a-fA-F]))+', '', text)  # Remove URLs
    text = re.sub(r'[\x00-\x1F\x7F-\x9F]', '', text)  # Remove control characters
    
    if aggressive:
        # More aggressive cleaning
        text = re.sub(r'[^\w\s.,!?\'"-]', '', text)  # Remove most special chars
        text = re.sub(r'\s+', ' ', text).strip()
    else:
        # Keep more special characters
        text = re.sub(r'[^\w\s.,!?@#$%^&*()\'"-]', '', text)
        text = re.sub(r'\s+', ' ', text).strip()
        
    text = text.replace('\ufffd', '')  # Remove replacement characters
    
    # Normalize whitespace and quotes
    text = re.sub(r'[\u2018\u2019]', "'", text)
    text = re.sub(r'[\u201C\u201D]', '"', text)
    text = re.sub(r'[\u2013\u2014]', '-', text)
    
    return text

def markdown_to_html(text: str, embed_images=False) -> str:
    """Convert markdown to HTML with enhanced table support and optional image embedding"""
    if not text:
        return ""
    
    extensions = ['fenced_code', 'tables', 'codehilite', 'nl2br', 'md_in_html']
    
    if embed_images:
        class ImageEmbedExtension(markdown.Extension):
            def extendMarkdown(self, md):
                md.inlinePatterns.register(ImageEmbedPattern(md.inlinePatterns['image_link']), 'image_embed', 150
        
        extensions.append(ImageEmbedExtension())
    
    return markdown.markdown(text, extensions=extensions)

class ImageEmbedPattern(markdown.inlinepatterns.ImageInlinePattern):
    def handleMatch(self, m):
        el = super().handleMatch(m)
        if el is not None:
            el.set('loading', 'lazy')
            el.set('class', 'markdown-image')
        return el

class ModelManager:
    _instance = None
    _lock = Lock()
    _shutdown = False

    def __new__(cls):  
        if cls._instance is None:  
            with cls._lock:  
                if cls._instance is None:  
                    cls._instance = super().__new__(cls)  
                    cls._instance._initialize()  
        return cls._instance  

    def _initialize(self):  
        self.models = {}  
        self.locks = {  
            "tts": Lock(),  
            "sd": Lock(),  
            "nlp": Lock(),  
            "whisper": Lock(),  
            "blip": Lock(),
            "llava": Lock(),
            "upscale": Lock(),
            "controlnet": Lock()
        }  
        
        self.config = load_config()
        self.model_queue = queue.PriorityQueue()
        self.preload_thread = Thread(target=self._preload_models, daemon=True)
        self.preload_thread.start()
        
        # Register cleanup
        atexit.register(self.cleanup)
        signal.signal(signal.SIGINT, self.signal_handler)
        signal.signal(signal.SIGTERM, self.signal_handler)

    def signal_handler(self, signum, frame):
        self._shutdown = True
        self.cleanup()
        sys.exit(0)

    def cleanup(self):
        """Enhanced cleanup with model unloading"""
        logger.info("Cleaning up models...")
        self._shutdown = True
        
        with self._lock:
            for name, model in self.models.items():
                try:
                    if hasattr(model, 'cpu'):
                        model.cpu()
                    del model
                except Exception as e:
                    logger.error(f"Failed to cleanup model {name}: {str(e)}")
            
            self.models.clear()
            clear_memory(aggressive=True)

    def _preload_models(self):  
        """Background model preloading with priority and memory management"""  
        try:
            logger.info("Starting prioritized model preloading...")
            
            # Load config-specified models first
            for model_type in self.config.get("system", {}).get("model_preload", []):
                if self._shutdown:
                    return
                
                try:
                    if model_type == "tts":
                        self.get_tts_model()  
                    elif model_type == "sd":
                        self.get_sd_pipeline()  
                    elif model_type == "nlp":
                        self.get_nlp_model()  
                    elif model_type == "whisper":
                        self.get_whisper_model()
                    elif model_type == "blip":
                        self.get_blip_model()
                    elif model_type == "llava":
                        self.get_llava_model()
                    elif model_type == "upscale":
                        self.get_upscale_pipeline()
                    elif model_type == "controlnet":
                        self.get_controlnet_pipeline()
                    
                    logger.info(f"Preloaded {model_type} model")
                except Exception as e:
                    logger.error(f"Failed to preload {model_type}: {str(e)}")
                
                if self.config.get("system", {}).get("auto_clear_memory", True):
                    clear_memory()
            
            # Process priority queue
            while not self._shutdown and not self.model_queue.empty():
                priority, (model_type, model_name) = self.model_queue.get()
                
                try:
                    if model_type == "llm":
                        self.get_llm(model_name)
                    elif model_type == "sd":
                        if model_name == "img2img":
                            self.get_img2img_pipeline()
                        elif model_name == "inpaint":
                            self.get_inpaint_pipeline()
                    
                    logger.info(f"Preloaded queued model {model_name}")
                except Exception as e:
                    logger.error(f"Failed to preload queued model {model_name}: {str(e)}")
                
                if self.config.get("system", {}).get("auto_clear_memory", True):
                    clear_memory()
                
                self.model_queue.task_done()
            
            logger.info("Model preloading completed")
        except Exception as e:
            logger.error(f"Model preloading failed: {str(e)}")
        finally:
            if not self._shutdown and self.config.get("system", {}).get("auto_clear_memory", True):
                clear_memory()

    def queue_model_load(self, model_type: str, model_name: str, priority: int = 1):
        """Queue a model for background loading with priority"""
        self.model_queue.put((priority, (model_type, model_name)))

    def get_llm(self, model_name=DEFAULT_LLM):  
        """Get LLM model with better error handling, fallback, and adaptive loading"""
        with self.locks["nlp"]:  
            cache_key = f"llm_{model_name}"
            
            if cache_key not in self.models:  
                try:
                    logger.info(f"Loading {model_name} LLM...")  
                    
                    # Adaptive loading based on system resources
                    params = LLM_MODELS.get(model_name, {}).get("params", {})
                    
                    if torch.cuda.is_available():
                        gpu_mem = torch.cuda.get_device_properties(0).total_memory / 1024**3
                        if gpu_mem < 16 and "70b" in model_name.lower():
                            raise ValueError(f"Insufficient GPU memory ({gpu_mem:.1f}GB) for {model_name}")
                    
                    self.models[cache_key] = Ollama(  
                        model=model_name,  
                        temperature=params.get("temperature", 0.7),
                        num_ctx=params.get("max_length", 4096),
                        num_gpu=1 if torch.cuda.is_available() else 0
                    )
                    
                    # Queue related models for background loading
                    if model_name == "llava-next":
                        self.queue_model_load("llava", "llava", priority=2)
                except Exception as e:  
                    logger.error(f"Failed to load {model_name}: {str(e)}")  
                    
                    # Try fallback models with resource awareness
                    fallback_order = sorted(
                        BACKUP_LLMS,
                        key=lambda x: (
                            -1 if x == model_name else 0,
                            LLM_MODELS.get(x, {}).get("params", {}).get("max_length", 4096)
                        )
                    )
                    
                    for fallback in fallback_order:
                        if fallback != model_name:
                            try:
                                logger.info(f"Trying fallback model: {fallback}")
                                fallback_params = LLM_MODELS.get(fallback, {}).get("params", {})
                                
                                # Check if we have enough resources for this fallback
                                if torch.cuda.is_available():
                                    required_mem = fallback_params.get("max_length", 4096) / 1024 * 2  # Approx GB needed
                                    gpu_mem = torch.cuda.get_device_properties(0).total_memory / 1024**3
                                    
                                    if required_mem > gpu_mem * 0.8:  # Leave 20% buffer
                                        logger.warning(f"Skipping {fallback} - requires ~{required_mem:.1f}GB, have {gpu_mem:.1f}GB")
                                        continue
                                
                                self.models[cache_key] = Ollama(
                                    model=fallback,
                                    temperature=fallback_params.get("temperature", 0.7),
                                    num_ctx=fallback_params.get("max_length", 4096),
                                    num_gpu=1 if torch.cuda.is_available() else 0
                                )
                                logger.info(f"Using fallback model: {fallback}")
                                break
                            except Exception as e:
                                logger.error(f"Fallback model {fallback} also failed: {str(e)}")
                    else:
                        logger.error("All models failed to load")
                        return None
            return self.models[cache_key]

class DocumentProcessor:
    @staticmethod
    def extract_text(file_path: str, max_pages=100) -> str:
        """Enhanced document text extraction with better file handling and page limits"""
        try:
            if not os.path.exists(file_path):
                raise FileNotFoundError(f"File not found: {file_path}")
                
            ext = os.path.splitext(file_path)[1].lower()
            file_size = os.path.getsize(file_path) / (1024 * 1024)  # MB
            
            # Security check for large files
            if file_size > 50:  # 50MB limit
                return f"File too large ({file_size:.1f}MB). Maximum allowed is 50MB."
            
            if ext == ".pdf":  
                return DocumentProcessor.extract_from_pdf(file_path, max_pages)  
            elif ext == ".docx":  
                return DocumentProcessor.extract_from_docx(file_path)  
            elif ext == ".txt":  
                return DocumentProcessor.extract_from_txt(file_path)  
            elif ext in [".csv", ".xlsx", ".xls"]:  
                return DocumentProcessor.extract_from_spreadsheet(file_path)  
            elif ext in [".pptx", ".ppt"]:  
                return DocumentProcessor.extract_from_presentation(file_path, max_pages)  
            elif ext in [".html", ".htm"]:
                return DocumentProcessor.extract_from_html(file_path)
            elif ext in [".epub"]:
                return DocumentProcessor.extract_from_epub(file_path, max_pages)
            elif ext in [".md", ".markdown"]:
                return DocumentProcessor.extract_from_markdown(file_path)
            elif ext in [".json"]:
                return DocumentProcessor.extract_from_json(file_path)
            elif ext in [".xml"]:
                return DocumentProcessor.extract_from_xml(file_path)
            elif ext in [".yaml", ".yml"]:
                return DocumentProcessor.extract_from_yaml(file_path)
            else:  
                logger.warning(f"Unsupported file format: {ext}")  
                return ""  
        except Exception as e:
            logger.error(f"Document processing failed: {str(e)}")
            return f"Error processing document: {str(e)}"

    @staticmethod  
    def extract_from_pdf(file_path: str, max_pages=100) -> str:  
        try:  
            text = ""  
            with fitz.open(file_path) as doc:  
                for i, page in enumerate(doc):  
                    if i >= max_pages:
                        text += f"\n\n[Document truncated after {max_pages} pages]"
                        break
                    text += page.get_text()  
            return clean_text(text)  
        except Exception as e:  
            logger.error(f"PDF processing error: {str(e)}")  
            return ""  

    @staticmethod  
    def extract_from_docx(file_path: str) -> str:  
        try:  
            doc = docx.Document(file_path)  
            full_text = []
            for para in doc.paragraphs:
                full_text.append(para.text)
            
            # Extract tables
            for table in doc.tables:
                for row in table.rows:
                    for cell in row.cells:
                        full_text.append(cell.text)
            
            return clean_text("\n".join(full_text))  
        except Exception as e:  
            logger.error(f"DOCX processing error: {str(e)}")  
            return ""  

    @staticmethod  
    def extract_from_txt(file_path: str) -> str:  
        try:  
            with open(file_path, 'r', encoding='utf-8', errors='ignore') as f:  
                return clean_text(f.read())  
        except Exception as e:  
            logger.error(f"TXT processing error: {str(e)}")  
            return ""  

    @staticmethod  
    def extract_from_spreadsheet(file_path: str) -> str:  
        try:  
            if file_path.endswith('.csv'):  
                df = pd.read_csv(file_path)  
            else:  # Excel  
                df = pd.read_excel(file_path)  
                
            # Handle multiple sheets
            if isinstance(df, dict):
                return "\n\n".join(
                    f"Sheet: {name}\n{sheet.to_markdown()}" 
                    for name, sheet in df.items()
                )
            return df.to_markdown()  
        except Exception as e:  
            logger.error(f"Spreadsheet processing error: {str(e)}")  
            return ""  

    @staticmethod  
    def extract_from_presentation(file_path: str, max_pages=50) -> str:  
        try:  
            text = ""  
            doc = fitz.open(file_path)  
            for i, page in enumerate(doc):  
                if i >= max_pages:
                    text += f"\n\n[Presentation truncated after {max_pages} slides]"
                    break
                text += page.get_text()  
            return clean_text(text)  
        except Exception as e:  
            logger.error(f"Presentation processing error: {str(e)}")  
            return ""
            
    @staticmethod
    def extract_from_html(file_path: str) -> str:
        try:
            with open(file_path, 'r', encoding='utf-8') as f:
                soup = BeautifulSoup(f.read(), 'html.parser')
                
                # Remove script and style elements
                for script in soup(["script", "style"]):
                    script.decompose()
                
                return clean_text(soup.get_text(separator='\n', strip=True))
        except Exception as e:
            logger.error(f"HTML processing error: {str(e)}")
            return ""
            
    @staticmethod
    def extract_from_epub(file_path: str, max_pages=100) -> str:
        try:
            text = ""
            with fitz.open(file_path) as doc:
                for i, page in enumerate(doc):
                    if i >= max_pages:
                        text += f"\n\n[EPUB truncated after {max_pages} pages]"
                        break
                    text += page.get_text()
            return clean_text(text)
        except Exception as e:
            logger.error(f"EPUB processing error: {str(e)}")
            return ""
            
    @staticmethod
    def extract_from_markdown(file_path: str) -> str:
        try:
            with open(file_path, 'r', encoding='utf-8') as f:
                return clean_text(f.read())
        except Exception as e:
            logger.error(f"Markdown processing error: {str(e)}")
            return ""
            
    @staticmethod
    def extract_from_json(file_path: str) -> str:
        try:
            with open(file_path, 'r', encoding='utf-8') as f:
                data = json.load(f)
                return json.dumps(data, indent=2)
        except Exception as e:
            logger.error(f"JSON processing error: {str(e)}")
            return ""
            
    @staticmethod
    def extract_from_xml(file_path: str) -> str:
        try:
            tree = ET.parse(file_path)
            root = tree.getroot()
            
            def recursive_parse(element, depth=0):
                text = f"{'  '*depth}{element.tag}"
                if element.text and element.text.strip():
                    text += f": {element.text.strip()}"
                text += "\n"
                
                for child in element:
                    text += recursive_parse(child, depth+1)
                    
                return text
                
            return recursive_parse(root)
        except Exception as e:
            logger.error(f"XML processing error: {str(e)}")
            return ""
            
    @staticmethod
    def extract_from_yaml(file_path: str) -> str:
        try:
            with open(file_path, 'r', encoding='utf-8') as f:
                data = yaml.safe_load(f)
                return yaml.dump(data, allow_unicode=True)
        except Exception as e:
            logger.error(f"YAML processing error: {str(e)}")
            return ""

    @staticmethod  
    def get_random_proxy() -> Optional[str]:  
        """Get a random proxy with validation and rotation"""
        if not PROXY_URLS:
            return None
            
        # Try proxies in random order until we find a working one
        tested_proxies = set()
        while len(tested_proxies) < len(PROXY_URLS):
            proxy = random.choice([p for p in PROXY_URLS if p not in tested_proxies])
            tested_proxies.add(proxy)
            
            try:
                # Test proxy with multiple sites
                test_urls = [
                    "https://www.google.com",
                    "https://api.ipify.org?format=json",
                    "https://httpbin.org/ip"
                ]
                
                for test_url in test_urls:
                    response = requests.get(
                        test_url, 
                        proxies={"http": proxy, "https": proxy}, 
                        timeout=10
                    )
                    if response.status_code != 200:
                        raise ConnectionError(f"Proxy failed with status {response.status_code}")
                
                # If all tests passed
                return proxy
            except:
                continue
                
        return None

    @staticmethod  
    @rate_limited(max_concurrent=5, per_second=3)  
    async def fetch_url(url: str, headers=None, proxies=None, timeout=15, use_selenium=False) -> Optional[str]:  
        """Enhanced async URL fetching with better error handling and retries"""
        max_retries = 3
        backoff_factor = 1
        
        async with aiohttp.ClientSession() as session:
            for attempt in range(max_retries):
                try:
                    if use_selenium:
                        return await WebScraper._fetch_with_selenium(url, timeout)
                    
                    async with session.get(
                        url, 
                        headers=headers or {"User-Agent": random.choice(WebScraper.USER_AGENTS)}, 
                        proxy=proxies,
                        timeout=aiohttp.ClientTimeout(total=timeout)
                    ) as response:
                        response.raise_for_status()
                        return await response.text()
                        
                except (aiohttp.ClientError, asyncio.TimeoutError) as e:
                    if attempt == max_retries - 1:
                        logger.warning(f"Failed to fetch {url} after {max_retries} attempts: {str(e)}")
                        return None
                    
                    sleep_time = backoff_factor * (2 ** attempt)
                    logger.info(f"Retry {attempt + 1} for {url} in {sleep_time}s...")
                    await asyncio.sleep(sleep_time)
        
        return None

    @staticmethod
    async def _fetch_with_selenium(url: str, timeout=15) -> Optional[str]:
        """Enhanced Selenium-based fetching with async support"""
        options = Options()
        options.add_argument("--headless")
        options.add_argument("--disable-gpu")
        options.add_argument("--no-sandbox")
        options.add_argument(f"user-agent={random.choice(WebScraper.USER_AGENTS)}")
        options.add_argument("--disable-blink-features=AutomationControlled")
        
        if is_tor_available() and ".onion" in url:
            options.add_argument(f'--proxy-server={TOR_PROXY}')
        
        # Set up Chrome in a separate thread
        def run_selenium():
            driver = webdriver.Chrome(options=options)
            try:
                driver.get(url)
                WebDriverWait(driver, timeout).until(
                    EC.presence_of_element_located((By.TAG_NAME, "body"))
                )
                return driver.page_source
            finally:
                driver.quit()
                
        loop = asyncio.get_event_loop()
        return await loop.run_in_executor(None, run_selenium)

        """Parallel web search with enhanced results and caching"""  
        all_data = ""  
        tasks = []  
        
        async with aiohttp.ClientSession() as session:
            for name, config in WebScraper.SEARCH_ENGINES.items():  
                if config["tor"] and not dark_mode:  
                    continue  
                    
                url = config["url"].format(query=requests.utils.quote(query), lang=lang)  
                tasks.append(asyncio.create_task(
                    WebScraper.fetch_url(  
                        url,  
                        {"User-Agent": random.choice(WebScraper.USER_AGENTS)},  
                        WebScraper.get_random_proxy(),
                        use_selenium=config["js"]
                    )
                ))  

            for task, (name, config) in zip(asyncio.as_completed(tasks), WebScraper.SEARCH_ENGINES.items()):  
                html = await task  
                if html:  
                    results = WebScraper.parse_results(html, config["parser"])  
                    if results:  
                        all_data += f"\n\n🌐 [{name} Results]:\n" + "\n".join(results)  
                          
        return clean_text(all_data[:MAX_CONTEXT_LENGTH])  # Use global constant

# ==================== ENHANCED LANGUAGE PROCESSING ====================

class LanguageProcessor:
    _instance = None
    _lock = Lock()
    
    def __new__(cls):
        if cls._instance is None:
            with cls._lock:
                if cls._instance is None:
                    cls._instance = super().__new__(cls)
                    cls._instance._initialize()
        return cls._instance
    
    def _initialize(self):
        self.translation_model = None
        self.translation_tokenizer = None
        self.embedding_model = None
        self.lock = Lock()
        
    def get_translation_model(self):
        """Lazy loading for translation model"""
        with self.lock:
            if self.translation_model is None:
                logger.info("Loading translation model...")
                model_name = "Helsinki-NLP/opus-mt-mul-en"
                self.translation_model = AutoModelForSeq2SeqLM.from_pretrained(
                    model_name,
                    cache_dir=str(MODEL_CACHE_DIR)
                ).to("cuda" if torch.cuda.is_available() else "cpu")
                self.translation_tokenizer = AutoTokenizer.from_pretrained(
                    model_name,
                    cache_dir=str(MODEL_CACHE_DIR)
                )
            return self.translation_model, self.translation_tokenizer
    
    def get_embedding_model(self):
        """Lazy loading for sentence embedding model"""
        with self.lock:
            if self.embedding_model is None:
                logger.info("Loading sentence embedding model...")
                self.embedding_model = SentenceTransformer(
                    'all-MiniLM-L6-v2',
                    cache_folder=str(MODEL_CACHE_DIR)
                ).to("cuda" if torch.cuda.is_available() else "cpu")
            return self.embedding_model

    @staticmethod
    def detect(text: str) -> str:
        """Enhanced language detection with better fallback"""
        if not text:
            return "en"
            
        try:
            # First try langdetect
            lang = detect(text)
            if lang in LANGUAGES:
                return lang
                
            # Then try character-based detection
            text = text.lower()
            char_ranges = {
                "hi": "अ-ह",
                "es": "áéíóúñ¿¡",
                "fr": "àâçéèêëîïôûùüÿ",
                "de": "äöüß",
                "ja": "あ-ん",
                "zh": "你好谢谢再见",
                "ru": "Привет спасибо до свидания",
                "ar": "مرحبا شكرا وداعا",
                "ko": "안녕하세요 감사합니다",
                "sv": "äöå",
                "pl": "ąćęłńóśźż",
                "th": "ก-ฮ",
                "vi": "ăâđêôơư",
                "el": "α-ω",
                "he": "א-ת"
            }
            
            for lang_code, chars in char_ranges.items():
                if any(char in text for char in chars):
                    return lang_code
                    
            return "en"
        except LangDetectException:
            return "en"

    async def translate(self, text: str, target_lang="en") -> str:  
        """Enhanced translation between supported languages"""
        if not text or target_lang not in LANGUAGES:  
            return text  
            
        try:
            # For languages we have special handling for
            if target_lang == "en":
                model, tokenizer = self.get_translation_model()
                
                inputs = tokenizer(text, return_tensors="pt", truncation=True, max_length=512)
                inputs = inputs.to(model.device)
                
                outputs = model.generate(**inputs)
                translated = tokenizer.decode(outputs[0], skip_special_tokens=True)
                
                return f"[Translated from {LanguageProcessor.detect(text)}]: {translated}"
            else:
                # For other languages, use a placeholder (in a real app, you'd use a translation API)
                return f"[Content would be translated to {LANGUAGES[target_lang]['name']}]: {text}"
        except Exception as e:
            logger.error(f"Translation failed: {str(e)}")
            return text

    def get_semantic_similarity(self, text1: str, text2: str) -> float:
        """Calculate semantic similarity between two texts"""
        model = self.get_embedding_model()
        
        embeddings = model.encode([text1, text2], convert_to_tensor=True)
        similarity = F.cosine_similarity(embeddings[0].unsqueeze(0), embeddings[1].unsqueeze(0))
        
        return similarity.item()

# ==================== ENHANCED MULTI-LLM ENGINE ====================

class MultiLLMEngine:
    def __init__(self):
        self.llm_cache = {}
        self.lock = Lock()
        self.memory = ConversationBufferMemory()
        self.task_history = []
        self.vector_memory = None
        self.embedding_model = None
        self.retriever = None
        self.adaptation_thread = None
        self.stop_adaptation = Event()
        logger.info("Initialized Enhanced Multi-LLM Engine")

        # Initialize vector memory
        self._init_vector_memory()
        
        # Start model adaptation thread
        self.start_adaptation()

    def _init_vector_memory(self):
        """Initialize vector store for long-term memory"""
        try:
            # Use FAISS for efficient similarity search
            self.embedding_model = SentenceTransformer(
                'all-MiniLM-L6-v2',
                cache_folder=str(MODEL_CACHE_DIR)
            
            index_path = KNOWLEDGE_BASE_DIR / "vector_memory.faiss"
            if index_path.exists():
                self.vector_memory = faiss.read_index(str(index_path))
            else:
                # Create new empty index
                dimension = self.embedding_model.get_sentence_embedding_dimension()
                self.vector_memory = faiss.IndexFlatL2(dimension)
                
            # Create retriever
            self.retriever = VectorStoreRetrieverMemory(
                vectorstore=self.vector_memory,
                embedding_model=self.embedding_model
            )
        except Exception as e:
            logger.error(f"Failed to initialize vector memory: {str(e)}")
            self.vector_memory = None
            self.retriever = None

    def save_vector_memory(self):
        """Save vector memory to disk"""
        if self.vector_memory:
            try:
                faiss.write_index(self.vector_memory, str(KNOWLEDGE_BASE_DIR / "vector_memory.faiss"))
            except Exception as e:
                logger.error(f"Failed to save vector memory: {str(e)}")

    def start_adaptation(self):
        """Start background model adaptation"""
        if self.adaptation_thread is None or not self.adaptation_thread.is_alive():
            self.stop_adaptation.clear()
            self.adaptation_thread = Thread(target=self._adapt_models, daemon=True)
            self.adaptation_thread.start()
            logger.info("Started model adaptation thread")

    def stop_adaptation_thread(self):
        """Stop the model adaptation thread"""
        self.stop_adaptation.set()
        if self.adaptation_thread and self.adaptation_thread.is_alive():
            self.adaptation_thread.join(timeout=5)
        self.save_vector_memory()

    def _adapt_models(self):
        """Continuously adapt models based on usage patterns"""
        while not self.stop_adaptation.is_set():
            try:
                # Analyze recent tasks for adaptation
                if self.task_history:
                    recent_tasks = self.task_history[-100:]  # Last 100 tasks
                    
                    # Cluster tasks by similarity
                    task_texts = [t["task"] for t in recent_tasks]
                    embeddings = self.embedding_model.encode(task_texts)
                    
                    # Simple k-means clustering
                    k = min(5, len(recent_tasks))
                    if k > 1:
                        kmeans = faiss.Kmeans(
                            self.embedding_model.get_sentence_embedding_dimension(),
                            k,
                            niter=20
                        )
                        kmeans.train(embeddings)
                        
                        # For each cluster, identify the best performing model
                        for i in range(k):
                            cluster_indices = np.where(kmeans.index.search(embeddings, 1)[1] == i
                            cluster_tasks = [recent_tasks[idx] for idx in cluster_indices[0]]
                            
                            # Find model with highest success rate in this cluster
                            model_performance = {}
                            for task in cluster_tasks:
                                model = task.get("model", DEFAULT_LLM)
                                success = task.get("success", True)
                                model_performance[model] = model_performance.get(model, 0) + (1 if success else 0)
                            
                            if model_performance:
                                best_model = max(model_performance.items(), key=lambda x: x[1])[0]
                                logger.info(f"Cluster {i} best model: {best_model}")
                                
                                # Could implement model switching logic here
                                
                # Sleep before next adaptation cycle
                self.stop_adaptation.wait(timeout=300)  # 5 minutes
                
            except Exception as e:
                logger.error(f"Model adaptation failed: {str(e)}")
                self.stop_adaptation.wait(timeout=60)  # Wait 1 minute before retry

    def get_llm(self, model_name=DEFAULT_LLM):  
        """Get LLM instance with better error handling and adaptive loading"""
        with self.lock:  
            cache_key = f"llm_{model_name}"
            
            if cache_key not in self.llm_cache:  
                llm = model_manager.get_llm(model_name)  
                if llm:  
                    self.llm_cache[cache_key] = llm  
                    
                    # Preload related models in background
                    if model_name in ["llava-next", "llava-1.6"]:
                        model_manager.queue_model_load("llava", "llava", priority=2)
                    
            return self.llm_cache.get(cache_key)  

    def select_best_model(self, prompt: str) -> str:
        """Select the best model based on prompt analysis"""
        # Simple keyword-based selection - could be enhanced with ML
        prompt_lower = prompt.lower()
        
        # Check for model-specific strengths
        for model, config in LLM_MODELS.items():
            for strength in config.get("strengths", []):
                if strength in prompt_lower:
                    return model
                    
        # Default to most capable model that fits in context
        max_length = len(prompt.split()) * 1.5  # Estimate token count
        
        suitable_models = [
            (model, config) 
            for model, config in LLM_MODELS.items()
            if config.get("params", {}).get("max_length", 4096) >= max_length
        ]
        
        if suitable_models:
            # Prefer models with larger context windows
            return max(suitable_models, key=lambda x: x[1]["params"]["max_length"])[0]
            
        return DEFAULT_LLM

    async def generate(self, prompt: str, model_name=None, fallback_models=BACKUP_LLMS, max_retries=3) -> str:  
        """Enhanced async text generation with better fallback and retry logic"""  
        if model_name is None:
            model_name = self.select_best_model(prompt)
            
        llm = self.get_llm(model_name)  
        models_tried = [model_name]  
        start_time = time.time()
        
        for attempt in range(max_retries):  
            try:  
                if not llm:  
                    raise ValueError(f"Model {model_name} not available")  
                      
                # Add relevant memory to prompt
                relevant_memory = self.retrieve_memory(prompt) if self.retriever else ""
                enhanced_prompt = f"""
                [Relevant Memory]:
                {relevant_memory}
                
                [Current Task]:
                {prompt}
                """
                
                response = await asyncio.get_event_loop().run_in_executor(
                    None,
                    lambda: llm(enhanced_prompt)
                )
                
                if response and len(response) > 30 and "sorry" not in response.lower():  
                    # Store successful generation in memory
                    self.store_memory(prompt, response)
                    
                    # Log successful generation
                    self.task_history.append({
                        "timestamp": datetime.now().isoformat(),
                        "model": model_name,
                        "task": prompt[:200],
                        "response": response[:200],
                        "success": True,
                        "latency": time.time() - start_time
                    })
                    
                    return clean_text(response)  
                raise ValueError("Insufficient response quality")  
            except Exception as e:  
                logger.warning(f"Attempt {attempt+1} with {model_name} failed: {str(e)}")  
                  
                # Try next available model  
                for fallback in fallback_models:  
                    if fallback not in models_tried:  
                        model_name = fallback  
                        llm = self.get_llm(model_name)  
                        models_tried.append(model_name)  
                        break  
                else:  
                    break  
        
        # Log failure
        self.task_history.append({
            "timestamp": datetime.now().isoformat(),
            "model": model_name,
            "task": prompt[:200],
            "error": str(e),
            "success": False,
            "latency": time.time() - start_time
        })
                      
        return "I'm experiencing technical difficulties. Please try again later."  

    def retrieve_memory(self, query: str, k=3) -> str:
        """Retrieve relevant memories for a query"""
        if not self.retriever:
            return ""
            
        try:
            results = self.retriever.get_relevant_documents(query, k=k)
            return "\n".join([doc.page_content for doc in results])
        except Exception as e:
            logger.error(f"Memory retrieval failed: {str(e)}")
            return ""

    def store_memory(self, input_text: str, output_text: str):
        """Store important information in long-term memory"""
        if not self.retriever:
            return
            
        try:
            # Only store if the information seems valuable
            if len(output_text) > 50 and not any(
                word in output_text.lower() 
                for word in ["error", "sorry", "can't", "don't know"]
            ):
                memory_text = f"Q: {input_text}\nA: {output_text}"
                self.retriever.add_documents([memory_text])
        except Exception as e:
            logger.error(f"Failed to store memory: {str(e)}")

    async def god_mode_generate(self, prompt: str, models=None) -> str:  
        """Enhanced parallel generation from multiple LLMs"""  
        models = models or ["deepseek-llm", "llama3-70b", "mixtral", "command-r-plus", "dolphin-2.9"]
        responses = []  
        
        # Run all generations in parallel
        tasks = [self.generate(prompt, model) for model in models]
        results = await asyncio.gather(*tasks, return_exceptions=True)
        
        for model, result in zip(models, results):
            if not isinstance(result, Exception):
                responses.append(f"🔥 {model.upper()}:\n{result}\n")
            else:
                logger.error(f"God mode generation failed for {model}: {str(result)}")
          
        return "\n".join(responses) if responses else "All models failed to respond"
        
    async def autonomous_agent(self, task_description: str) -> str:
        """Enhanced autonomous task execution with memory"""
        prompt_template = """
        You are an advanced autonomous AI agent. Your task is to:
        1. Thoroughly analyze the task: {task}
        2. Break it down into logical subtasks
        3. Determine required resources/tools
        4. Execute subtasks efficiently
        5. Synthesize results
        6. Return comprehensive final output
        
        Context:
        - Current time: {time}
        - Relevant memories: {memory}
        - Previous steps: {history}
        
        Current step: {step}
        """
        
        prompt = PromptTemplate(
            input_variables=["task", "time", "memory", "history", "step"],
            template=prompt_template
        )
        
        llm_chain = LLMChain(
            llm=self.get_llm(DEFAULT_LLM),
            prompt=prompt,
            memory=self.memory
        )
        
        try:
            # Retrieve relevant memories
            memory_context = self.retrieve_memory(task_description)
            
            result = await asyncio.get_event_loop().run_in_executor(
                None,
                lambda: llm_chain.run({
                    "task": task_description,
                    "time": datetime.now().isoformat(),
                    "memory": memory_context,
                    "history": self.memory.buffer,
                    "step": "Initial analysis"
                })
            )
            
            # Log the task
            self.task_history.append({
                "timestamp": datetime.now().isoformat(),
                "task": task_description,
                "result": result[:500] + "..." if len(result) > 500 else result,
                "type": "autonomous"
            })
            
            return result
        except Exception as e:
            logger.error(f"Autonomous agent failed: {str(e)}")
            return f"Task execution failed: {str(e)}"

        num_inference_steps=50,
        guidance_scale=9.0
    ) -> Optional[Image.Image]:
        """Advanced image generation with ControlNet support"""
        try:
            if control_image and control_type:
                # ControlNet generation
                pipe = model_manager.get_controlnet_pipeline()
                
                # Convert control image based on type
                if control_type == "canny":
                    control_array = np.array(control_image)
                    control_array = cv2.Canny(control_array, 100, 200)
                    control_image = Image.fromarray(control_array)
                
                image = pipe(
                    prompt=prompt,
                    negative_prompt=negative_prompt,
                    image=control_image,
                    num_inference_steps=num_inference_steps,
                    guidance_scale=guidance_scale,
                    controlnet_conditioning_scale=strength
                ).images[0]
            elif mask_image and init_image:
                # Inpainting mode
                pipe = model_manager.get_inpaint_pipeline()
                image = pipe(
                    prompt=prompt,
                    negative_prompt=negative_prompt,
                    image=Image.fromarray(init_image) if isinstance(init_image, np.ndarray) else init_image,
                    mask_image=Image.fromarray(mask_image) if isinstance(mask_image, np.ndarray) else mask_image,
                    strength=strength,
                    guidance_scale=guidance_scale,
                    num_inference_steps=num_inference_steps
                ).images[0]
            elif init_image:
                # Image-to-image mode
                pipe = model_manager.get_img2img_pipeline()
                image = pipe(
                    prompt=prompt,
                    negative_prompt=negative_prompt,
                    image=Image.fromarray(init_image) if isinstance(init_image, np.ndarray) else init_image,
                    strength=strength,
                    guidance_scale=guidance_scale,
                    num_inference_steps=num_inference_steps
                ).images[0]
            else:
                # Text-to-image mode
                pipe = model_manager.get_sd_pipeline()
                image = pipe(
                    prompt=prompt,
                    negative_prompt=negative_prompt or "text, signature, watermark, deformed, blurry, nsfw",
                    num_inference_steps=num_inference_steps,
                    guidance_scale=guidance_scale,
                    width=1024,
                    height=1024
                ).images[0]

            # Post-processing  
            return ImageProcessor.enhance_quality(image)  
        except torch.cuda.OutOfMemoryError:  
            torch.cuda.empty_cache()  
            return ImageProcessor.generate(
                prompt, 
                negative_prompt, 
                init_image, 
                mask_image,
                control_image,
                control_type,
                strength,
                num_inference_steps,
                guidance_scale
            )  
        except Exception as e:  
            logger.error(f"Image generation failed: {str(e)}")  
            return None  

    @staticmethod  
    def upscale(image: Image.Image, factor=4) -> Image.Image:  
        """Enhanced image upscaling with better error handling"""
        try:
            # Try diffusion upscaling first
            pipe = model_manager.get_upscale_pipeline()
            upscaled_image = pipe(
                prompt="high quality, detailed, sharp",
                image=image,
                num_inference_steps=50
            ).images[0]
            
            # Fallback to traditional methods if needed
            if upscaled_image.size[0] < image.size[0] * factor:
                logger.warning("Diffusion upscaling didn't achieve desired size, using traditional methods")
                upscaled_image = image.resize(
                    (image.width*factor, image.height*factor), 
                    Image.LANCZOS
                )
                
            return upscaled_image
        except Exception as e:
            logger.warning(f"Upscaling failed, using traditional method: {str(e)}")
            return image.resize((image.width*factor, image.height*factor), Image.LANCZOS)  

    @staticmethod
    def enhance_quality(image: Image.Image) -> Image.Image:
        """Apply quality enhancement to image with better error handling"""
        try:
            # First upscale
            image = ImageProcessor.upscale(image)
            
            # Convert to array for OpenCV processing
            img_array = np.array(image)
            
            # Convert to LAB color space for better contrast enhancement
            lab = cv2.cvtColor(img_array, cv2.COLOR_BGR2LAB)
            l, a, b = cv2.split(lab)
            
            # CLAHE (Contrast Limited Adaptive Histogram Equalization)
            clahe = cv2.createCLAHE(clipLimit=3.0, tileGridSize=(8,8))
            l = clahe.apply(l)
            
            # Merge channels and convert back to BGR
            lab = cv2.merge((l,a,b))
            img_array = cv2.cvtColor(lab, cv2.COLOR_LAB2BGR)
            
            # Sharpening with unsharp mask
            blurred = cv2.GaussianBlur(img_array, (0,0), 3)
            img_array = cv2.addWeighted(img_array, 1.5, blurred, -0.5, 0)
            
            # Color correction
            img_array = cv2.cvtColor(img_array, cv2.COLOR_BGR2RGB)
            
            return Image.fromarray(img_array)
        except Exception as e:
            logger.error(f"Image enhancement failed: {str(e)}")
            return image

    @staticmethod  
    def process(image: Image.Image, operation: str) -> Image.Image:  
        """Enhanced image processing with more operations"""
        try:
            img = np.array(image)  
              
            if operation == "edge_detection":  
                result = cv2.Canny(img, 100, 200)  
            elif operation == "cartoonify":  
                # Convert to grayscale
                gray = cv2.cvtColor(img, cv2.COLOR_BGR2GRAY)  
                gray = cv2.medianBlur(gray, 5)  
                
                # Detect edges
                edges = cv2.adaptiveThreshold(gray, 255, cv2.ADAPTIVE_THRESH_MEAN_C, 
                                            cv2.THRESH_BINARY, 9, 9)  
                
                # Color quantization
                color = cv2.bilateralFilter(img, 9, 300, 300)  
                
                # Combine edges with quantized color
                result = cv2.bitwise_and(color, color, mask=edges)  
            elif operation == "grayscale":  
                result = cv2.cvtColor(img, cv2.COLOR_BGR2GRAY)  
            elif operation == "enhance":  
                # Contrast Limited Adaptive Histogram Equalization  
                lab = cv2.cvtColor(img, cv2.COLOR_BGR2LAB)  
                l, a, b = cv2.split(lab)  
                clahe = cv2.createCLAHE(clipLimit=3.0, tileGridSize=(8,8))  
                l = clahe.apply(l)  
                lab = cv2.merge((l,a,b))  
                result = cv2.cvtColor(lab, cv2.COLOR_LAB2BGR)  
            elif operation == "super_resolution":
                result = ImageProcessor.upscale(Image.fromarray(img))
                return result
            elif operation == "oil_painting":
                result = cv2.xphoto.oilPainting(img, 7, 1)
            elif operation == "pencil_sketch":
                gray = cv2.cvtColor(img, cv2.COLOR_BGR2GRAY)
                inverted = cv2.bitwise_not(gray)
                blurred = cv2.GaussianBlur(inverted, (21, 21), 0)
                result = cv2.divide(gray, 255 - blurred, scale=256)
            elif operation == "watercolor":
                # Apply bilateral filter multiple times
                result = cv2.bilateralFilter(img, 9, 75, 75)
                result = cv2.bilateralFilter(result, 9, 75, 75)
            elif operation == "hdr":
                # Tone mapping for HDR effect
                tonemap = cv2.createTonemapReinhard(1.5, 0.5, 0.5, 0.5)
                result = tonemap.process(img.astype(np.float32) * 255
                result = result.astype(np.uint8)
            else:  
                result = img  
                  
            return Image.fromarray(result)
        except Exception as e:
            logger.error(f"Image processing failed: {str(e)}")
            return image

    @staticmethod  
    def caption(image: Image.Image) -> str:  
        """Enhanced image captioning with better error handling"""
        try:  
            processor, model = model_manager.get_blip_model()  
            inputs = processor(images=image, return_tensors="pt").to("cuda" if torch.cuda.is_available() else "cpu")  
            outputs = model.generate(**inputs)  
            return processor.decode(outputs[0], skip_special_tokens=True)  
        except Exception as e:  
            logger.error(f"Image captioning failed: {str(e)}")  
            return "Could not generate caption for this image"
            
    @staticmethod
    def visual_qa(image: Image.Image, question: str) -> str:
        """Enhanced visual question answering"""
        try:
            pipe = model_manager.get_llava_model()
            result = pipe(image, question)
            return result[0]['generated_text']
        except Exception as e:
            logger.error(f"Visual QA failed: {str(e)}")
            return "Could not answer question about this image"
            
    @staticmethod
    def extract_text(image: Image.Image) -> str:
        """Enhanced OCR text extraction from images"""
        try:
            # Use pytesseract if available
            try:
                import pytesseract
                return pytesseract.image_to_string(np.array(image))
            except ImportError:
                logger.warning("pytesseract not available, using fallback OCR")
                
                # Fallback to easyocr if available
                try:
                    import easyocr
                    reader = easyocr.Reader(['en'])
                    results = reader.readtext(np.array(image))
                    return "\n".join([res[1] for res in results])
                except ImportError:
                    logger.warning("No OCR libraries available")
                    return "Install pytesseract or easyocr for OCR functionality"
        except Exception as e:
            logger.error(f"OCR failed: {str(e)}")
            return "Could not extract text from image"

# ==================== ENHANCED AUDIO PROCESSING ====================

class AudioProcessor:
    @staticmethod
    def text_to_speech(
        text: str, 
        lang="en", 
        speed=1.0, 
        personality="professional",
        voice_sample=None
    ) -> Optional[str]:
        """Enhanced TTS with voice cloning and effects"""
        try:
            if not text:
                return None

            # Truncate very long texts
            if len(text) > 1000:
                text = text[:1000] + "... [truncated]"

            timestamp = str(int(time.time()))  
            output_file = str(TEMP_DIR / f"reply_{timestamp}.wav")  
            
            # Apply personality effects  
            personality_config = PERSONALITIES.get(personality, {})
            if personality == "chad":  
                text = f"Listen up, bro. {text} 🏋️♂️"  
            elif personality == "sarcastic":  
                text = f"Oh great, another request. {text} 😏"  
            elif personality == "dark":  
                text = f"In this meaningless void... {text} 💀"  
            elif personality == "hacker":
                text = f"[HACKER VOICE ACTIVATED] {text} 01010100 01110010 01100001 01101110 01110011 01101101 01101001 01110011 01110011 01101001 01101111 01101110 00100000 01100011 01101111 01101101 01110000 01101100 01100101 01110100 01100101"
            elif personality == "god":
                text = f"BEHOLD MORTALS! {text.upper()}"
            elif personality == "pirate":
                text = f"Arrr matey! {text} ☠️"
            elif personality == "wizard":
                text = f"By the ancient spells... {text} ✨"
            elif personality == "detective":
                text = f"Elementary, my dear Watson. {text} 🕵️‍♂️"
            elif personality == "chef":
                text = f"Today's special is... {text} 👨‍🍳"
            elif personality == "titan":
                text = f"[TITAN MODE ACTIVATED] {text} ⚡"

            tts = model_manager.get_tts_model()  
            
            # Use provided voice sample or default for language
            speaker_wav = None
            if voice_sample and os.path.exists(voice_sample):
                speaker_wav = voice_sample
            else:
                default_voice = VOICE_SAMPLES_DIR / f"{lang}_voice.wav"
                if default_voice.exists():
                    speaker_wav = str(default_voice)
                else:
                    speaker_wav = str(VOICE_SAMPLES_DIR / "en_voice.wav")

            tts.tts_to_file(  
                text=text,  
                file_path=output_file,  
                speaker_wav=speaker_wav,  
                language=lang,  
                speed=speed  
            )  
              
            AudioProcessor.enhance_audio(output_file)  
            return output_file  
        except Exception as e:  
            logger.error(f"TTS failed: {str(e)} - Falling back to pyttsx3")  
            try:  
                engine = pyttsx3.init()  
                engine.setProperty('rate', 150 * speed)  
                engine.save_to_file(text[:500], output_file)  
                engine.runAndWait()  
                return output_file  
            except Exception as e:  
                logger.error(f"pyttsx3 fallback failed: {str(e)}")  
                return None  

    @staticmethod  
    def enhance_audio(input_file: str):
        """Enhanced audio processing with effects"""
        try:
            sound = AudioSegment.from_file(input_file)
            
            # Noise reduction
            sound = sound.low_pass_filter(3000).high_pass_filter(200)
            
            # Normalize volume
            sound = sound.normalize()
            
            # Compress dynamic range
            sound = sound.compress_dynamic_range()
            
            # Add slight reverb for more natural sound
            sound = sound.fade_in(50).fade_out(50)
            
            # Export enhanced audio
            sound.export(input_file, format="wav", parameters=["-acodec", "pcm_s16le"])
        except Exception as e:
            logger.warning(f"Audio enhancement failed: {str(e)}")

    @staticmethod  
    def speech_to_text(audio_file: str, language="auto") -> str:  
        """Enhanced speech recognition with language detection"""
        try:  
            model = model_manager.get_whisper_model()  
            
            # Load audio with noise reduction
            audio = whisper.load_audio(audio_file)
            audio = whisper.pad_or_trim(audio)
            
            # Detect language if auto
            if language == "auto":
                mel = whisper.log_mel_spectrogram(audio).to(model.device)
                _, probs = model.detect_language(mel)
                language = max(probs, key=probs.get)
                logger.info(f"Detected language: {language} with probability {probs[language]:.2f}")
            
            # Enhanced processing
            result = model.transcribe(
                audio,
                temperature=0.2,
                best_of=5,
                beam_size=5,
                suppress_tokens=[-1],
                initial_prompt="Transcribe accurately with punctuation and capitalization.",
                language=language if language != "auto" else None
            )
            
            return clean_text(result["text"])  
        except Exception as e:  
            logger.error(f"Speech recognition failed: {str(e)}")  
            return "Could not understand audio"

    @staticmethod
    def clone_voice(reference_audio: str, output_dir=VOICE_SAMPLES_DIR, lang="en") -> bool:
        """Create a voice clone from reference audio"""
        try:
            if not os.path.exists(reference_audio):
                raise FileNotFoundError(f"Audio file not found: {reference_audio}")
                
            # Create output directory if needed
            output_dir.mkdir(exist_ok=True)
            
            # Process the audio
            output_file = output_dir / f"{lang}_voice.wav"
            
            # Load and enhance the audio
            audio = AudioSegment.from_file(reference_audio)
            audio = audio.set_channels(1).set_frame_rate(22050)
            audio.export(output_file, format="wav")
            
            logger.info(f"Voice sample saved to {output_file}")
            return True
        except Exception as e:
            logger.error(f"Voice cloning failed: {str(e)}")
            return False

                module = importlib.util.module_from_spec(spec)
                spec.loader.exec_module(module)
                
                # Initialize plugin
                if hasattr(module, "Plugin"):
                    self.plugins[plugin_name] = module.Plugin(self)
                    logger.info(f"Loaded plugin: {plugin_name}")

    async def execute_task(self, task_description: str) -> str:  
        """Enhanced autonomous task execution with security checks"""
        if self.security_mode:
            # Check for dangerous commands
            dangerous_commands = [
                "delete", "remove", "format", "shutdown", 
                "restart", "kill", "stop", "rm -rf", 
                "drop table", "insert into", "update", 
                "alter table", "chmod 777", "sudo"
            ]
            
            if any(cmd in task_description.lower() for cmd in dangerous_commands):
                return "⚠️ Security mode prevents execution of potentially dangerous tasks"
            
            # Check for dark web access
            if "dark web" in task_description.lower() and not self.dark_mode:
                return "Dark web access requires dark mode to be enabled"
            
        task_id = f"task_{int(time.time())}"  
        self.active_tasks.append(task_id)  
        
        if self.learning_mode:
            self.learn_from_task(task_description)
          
        # Enhanced task execution with different capabilities
        if "search" in task_description.lower() or "find" in task_description.lower():  
            result = f"🔍 Search completed for: {task_description}\n{await WebScraper.search(task_description, dark_mode=self.dark_mode)[:2000]}..."
        elif "scrape" in task_description.lower() or "extract" in task_description.lower():  
            result = f"🕷️ Web scraping completed: {task_description}\n{await self.enhanced_scrape(task_description)}"
        elif "generate image" in task_description.lower() or "create picture" in task_description.lower():
            prompt = task_description.replace("generate image", "").replace("create picture", "").strip()
            image = ImageProcessor.generate(prompt)
            if image:
                image_path = str(TEMP_DIR / f"generated_{task_id}.png")
                image.save(image_path)
                result = f"🎨 Image generated and saved to {image_path}"
            else:
                result = "Failed to generate image"
        elif "execute code" in task_description.lower():
            if self.security_mode:
                result = "⚠️ Security mode prevents code execution"
            else:
                result = await self.execute_code(task_description)
        elif "remember" in task_description.lower():
            key = task_description.split("remember")[1].split("as")[0].strip()
            value = task_description.split("as")[1].strip()
            self.memory[key] = value
            result = f"🧠 Remembered '{key}' as '{value}'"
        elif "what do you know about" in task_description.lower():
            key = task_description.split("about")[1].strip()
            result = self.memory.get(key, f"I don't know anything about {key}")
        elif "voice clone" in task_description.lower():
            if "current voice" in task_description.lower() and self.current_voice_sample:
                success = AudioProcessor.clone_voice(self.current_voice_sample)
                result = "Voice cloned successfully!" if success else "Voice cloning failed"
            else:
                result = "No voice sample set for cloning"
        else:  
            result = await self.llm_engine.autonomous_agent(task_description)
              
        self.active_tasks.remove(task_id)  
        return result

    async def enhanced_scrape(self, url: str) -> str:
        """Enhanced web scraping with JavaScript rendering and security checks"""
        if self.security_mode:
            # Check for dark web domains
            if ".onion" in url.lower() and not self.dark_mode:
                return "Dark web scraping requires dark mode to be enabled"
            
            # Check for suspicious domains
            suspicious_domains = [
                ".exe", ".bat", ".sh", ".php", 
                "hack", "exploit", "malware"
            ]
            
            if any(domain in url.lower() for domain in suspicious_domains):
                return "Security blocked access to suspicious domain"
            
        try:
            options = Options()
            options.add_argument("--headless")
            options.add_argument("--disable-gpu")
            options.add_argument("--no-sandbox")
            options.add_argument("--disable-dev-shm-usage")
            options.add_argument(f"user-agent={random.choice(WebScraper.USER_AGENTS)}")
            
            if self.dark_mode and ".onion" in url.lower():
                if not is_tor_available():
                    return "Tor proxy not available for dark web access"
                options.add_argument(f'--proxy-server={TOR_PROXY}')
            
            # Run Selenium in a separate thread
            def run_selenium():
                driver = webdriver.Chrome(options=options)
                try:
                    driver.get(url if url.startswith("http") else f"https://{url}")
                    
                    # Wait for page to load
                    WebDriverWait(driver, 10).until(
                        EC.presence_of_element_located((By.TAG_NAME, "body"))
                    
                    # Get all text content
                    text = driver.find_element(By.TAG_NAME, "body").text
                    
                    # Take screenshot
                    screenshot_path = str(TEMP_DIR / f"screenshot_{int(time.time())}.png")
                    driver.save_screenshot(screenshot_path)
                    
                    return text, screenshot_path
                finally:
                    driver.quit()
                    
            loop = asyncio.get_event_loop()
            text, screenshot_path = await loop.run_in_executor(None, run_selenium)
            
            return f"Scraped text (first 2000 chars): {text[:2000]}...\nScreenshot saved to {screenshot_path}"
        except Exception as e:
            return f"Scraping failed: {str(e)}"

    async def execute_code(self, code_description: str) -> str:
        """Enhanced code execution with security checks"""
        if self.security_mode:
            return "⚠️ Security mode prevents code execution"
            
        try:
            # Generate code from description
            code = await self.llm_engine.generate(
                f"Generate Python code to: {code_description}. "
                "Return only the code block with no explanations."
            )
            
            # Clean the code
            code = re.sub(r'```python|```', '', code).strip()
            
            # Security check - block dangerous operations
            dangerous_patterns = [
                r'os\.system\(',
                r'subprocess\.(run|call|Popen)\(',
                r'eval\(',
                r'exec\(',
                r'open\(',
                r'__import__\(',
                r'pickle\.load\(',
                r'requests\.(get|post)\(',
                r'socket\.',
                r'shutil\.',
                r'rmtree\(',
                r'write\('
            ]
            
            if any(re.search(pattern, code) for pattern in dangerous_patterns):
                return "⚠️ Code contains potentially dangerous operations and was blocked"
            
            # Create a restricted globals dictionary
            safe_globals = {
                '__builtins__': {
                    'range': range,
                    'len': len,
                    'str': str,
                    'int': int,
                    'float': float,
                    'list': list,
                    'dict': dict,
                    'tuple': tuple,
                    'set': set,
                    'bool': bool,
                    'print': print,
                    'sum': sum,
                    'min': min,
                    'max': max,
                    'abs': abs,
                    'round': round
                },
                'math': __import__('math'),
                'datetime': __import__('datetime'),
                'random': __import__('random'),
                'json': __import__('json'),
                're': __import__('re')
            }
            
            # Execute in a safe environment
            local_vars = {}
            exec(code, safe_globals, local_vars)
            
            # Capture output
            output = "\n".join([
                str(v) for k, v in local_vars.items() 
                if not k.startswith('_') and not callable(v)
            ])
            
            return f"Code executed successfully:\n```python\n{code}\n```\nOutput:\n{output}"
        except Exception as e:
            return f"Code execution failed: {str(e)}"

    async def process_query(
        self, 
        input_text: str, 
        lang="auto", 
        gen_image=False, 
        input_image=None, 
        mask_image=None, 
        img_strength=0.75, 
        doc_file=None, 
        audio_input=None, 
        img_operation=None, 
        visual_question=None,
        control_image=None,
        control_type="canny"
    ) -> Dict[str, Union[str, Image.Image, None]]:  
        """Enhanced main processing pipeline with better error handling"""
        start_time = time.time()  
        results = {  
            "text": "",  
            "voice": None,  
            "image": None,  
            "summary": "",  
            "html": "",  
            "caption": "",
            "visual_qa": ""
        }  

        try:  
            # Process audio input  
            if audio_input:  
                input_text = AudioProcessor.speech_to_text(audio_input)  
                if not input_text:  
                    input_text = "Could not understand audio input"  

            # Process document input  
            doc_text = ""  
            if doc_file:  
                doc_text = DocumentProcessor.extract_text(doc_file.name)  
                input_text = f"{input_text}\n\n[Document Content]:\n{doc_text[:3000]}"  

            # Auto-detect language  
            if lang == "auto":  
                lang = LanguageProcessor.detect(input_text)  

            # Visual question answering
            if input_image and visual_question:
                results["visual_qa"] = ImageProcessor.visual_qa(input_image, visual_question)
                return results

            # Image processing mode  
            if input_image and img_operation:  
                results["image"] = ImageProcessor.process(input_image, img_operation)  
                results["caption"] = ImageProcessor.caption(results["image"])  
                return results  

            # Image generation mode  
            if gen_image and input_text:  
                results["image"] = ImageProcessor.generate(  
                    prompt=input_text,  
                    init_image=input_image,  
                    mask_image=mask_image,  
                    control_image=control_image,
                    control_type=control_type,
                    strength=img_strength  
                )  
                results["caption"] = ImageProcessor.caption(results["image"])  
                return results  

            # Web scraping for context  
            context = await WebScraper.search(input_text, lang, self.dark_mode)  

            # Generate AI response  
            prompt = f"""  
            [User Query]: {input_text}  
            [Language]: {lang}  
            [Personality]: {self.personality}  
            [Web Context]: {context[:3000]}  
              
            Instructions:  
            1. Analyze the query and context thoroughly  
            2. Provide a detailed, well-structured response  
            3. Include relevant examples and actionable advice  
            4. Format with markdown (headings, lists, etc.)  
            5. Respond in the user's language ({lang})  
            6. Add personality: {PERSONALITIES[self.personality]['tone']}  
            7. If technical, include code examples  
            8. If creative, be imaginative  
            9. If factual, cite sources  
            10. If ambiguous, ask clarifying questions  
              
            Response:  
            """  

            if self.god_mode:  
                response = await self.llm_engine.god_mode_generate(prompt)  
            else:  
                response = await self.llm_engine.generate(prompt)  

            results["text"] = response  

            # Generate summary  
            summary_prompt = f"Summarize this in 3 bullet points in {lang} with {self.personality} tone:\n{response}"  
            results["summary"] = await self.llm_engine.generate(summary_prompt)  

            # Convert to HTML  
            results["html"] = markdown_to_html(response)  

            # Voice response  
            if lang in LANGUAGES:  
                voice_sample = self.current_voice_sample if self.voice_clone_mode else None
                results["voice"] = AudioProcessor.text_to_speech(  
                    response if len(response) < 500 else results["summary"],  
                    lang=lang,  
                    personality=self.personality,
                    voice_sample=voice_sample
                )  

            logger.info(f"Processed query in {time.time()-start_time:.2f}s")  
            return results  

        except Exception as e:  
            logger.error(f"Processing failed: {str(e)}")  
            results["text"] = "An error occurred while processing your request. Please try again."  
            return results

            logger.error(f"Failed to launch interface: {str(e)}")
            raise
        finally:
            self.running = False
            self.jarvis.llm_engine.stop_adaptation_thread()
            if self.jarvis.file_watcher:
                self.jarvis.file_watcher.stop()
                self.jarvis.file_watcher.join()
            if self.jarvis.api_server:
                self.jarvis.api_server.should_exit = True
